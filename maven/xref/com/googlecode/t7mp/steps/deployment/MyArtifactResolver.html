16  package com.googlecode.t7mp.steps.deployment;
17  
18  import java.util.List;
19  
20  import org.apache.maven.artifact.Artifact;
21  import org.apache.maven.artifact.factory.ArtifactFactory;
22  import org.apache.maven.artifact.repository.ArtifactRepository;
23  import org.apache.maven.artifact.repository.ArtifactRepositoryPolicy;
24  import org.apache.maven.artifact.repository.DefaultArtifactRepository;
25  import org.apache.maven.artifact.repository.layout.DefaultRepositoryLayout;
26  import org.apache.maven.artifact.resolver.ArtifactNotFoundException;
27  import org.apache.maven.artifact.resolver.ArtifactResolutionException;
28  import org.apache.maven.artifact.resolver.ArtifactResolver;
29  import org.apache.maven.artifact.versioning.VersionRange;
30  import org.apache.maven.plugin.MojoExecutionException;
31  
32  import com.googlecode.t7mp.AbstractT7Mojo;
33  
34  /**
35   * Uses Maven-API to resolve the Artifacts.
36   * 
37   *
38   */
39  public class MyArtifactResolver {
40  
41      private ArtifactResolver resolver;
42      private ArtifactFactory factory;
43      private ArtifactRepository local;
44      private List<ArtifactRepository> remoteRepositories;
45      private boolean resolveAllways = false;
46  
47      public MyArtifactResolver(AbstractT7Mojo t7Mojo) {
48          this.remoteRepositories = t7Mojo.getRemoteRepos();
49          this.local = t7Mojo.getLocal();
50          this.resolver = t7Mojo.getResolver();
51          this.factory = t7Mojo.getFactory();
52          this.resolveAllways = t7Mojo.isResolverUpdateSnapshotsAllways();
53      }
54  
55      /**
56       * resolves an Artifact from the repositorys.
57       * 
58       * @param groupId - Artifact GroupId
59       * @param artifactId - Artifact Id
60       * @param version - Artifact Version
61       * @param classifier - Artifact Version
62       * @param type - Artifact Type
63       * @param scope - Artifact Scope
64       * @return
65       * @throws MojoExecutionException
66       */
67      public Artifact resolve(String groupId, String artifactId, String version, String classifier, String type, String scope)
68              throws MojoExecutionException {
69          if (version.endsWith("SNAPSHOT")) {
70              this.remoteRepositories.add(createStagingRepository());
71              this.remoteRepositories.add(createSnapshotsRepository());
72          }
73          Artifact artifact = factory.createDependencyArtifact(groupId, artifactId,
74                  VersionRange.createFromVersion(version), type, classifier, Artifact.SCOPE_COMPILE);
75          try {
76              resolver.resolve(artifact, remoteRepositories, local);
77          } catch (ArtifactResolutionException e) {
78              throw new MojoExecutionException(e.getMessage(), e);
79          } catch (ArtifactNotFoundException e) {
80              throw new MojoExecutionException(e.getMessage(), e);
81          }
82          return artifact;
83      }
84  
85      private ArtifactRepository createSnapshotsRepository() {
86          ArtifactRepository repository = new DefaultArtifactRepository("t7mp.apache.tomcat.snapshots",
87                  "http://people.apache.org/repo/m2-snapshot-repository", new DefaultRepositoryLayout(),
88                  createSnapshotPolicy(), createRelasesPolicy());
89          return repository;
90      }
91  
92      private ArtifactRepository createStagingRepository() {
93          ArtifactRepository repository = new DefaultArtifactRepository("t7mp.apache.tomcat.dev",
94                  "http://tomcat.apache.org/dev/dist/m2-repository", new DefaultRepositoryLayout(),
95                  createSnapshotPolicy(), createRelasesPolicy());
96          return repository;
97      }
98  
99      private ArtifactRepositoryPolicy createSnapshotPolicy() {
100         String updatePolicy = ArtifactRepositoryPolicy.UPDATE_POLICY_ALWAYS;
101         if (!resolveAllways) {
102             updatePolicy = ArtifactRepositoryPolicy.UPDATE_POLICY_DAILY;
103         }
104         return new ArtifactRepositoryPolicy(true, updatePolicy, ArtifactRepositoryPolicy.CHECKSUM_POLICY_WARN);
105     }
106 
107     private ArtifactRepositoryPolicy createRelasesPolicy() {
108         return new ArtifactRepositoryPolicy(false, ArtifactRepositoryPolicy.UPDATE_POLICY_DAILY,
109                 ArtifactRepositoryPolicy.CHECKSUM_POLICY_WARN);
110     }
111 
112     public Artifact resolveJar(String groupId, String artifactId, String version, String classifier) throws MojoExecutionException {
113         return resolve(groupId, artifactId, version, classifier, "jar", Artifact.SCOPE_COMPILE);
114     }
115 
116     public Artifact resolveWar(String groupId, String artifactId, String version, String classifier) throws MojoExecutionException {
117         return resolve(groupId, artifactId, version, classifier, "war", Artifact.SCOPE_COMPILE);
118     }
