29  public class TomcatArtifactTest {
30  
31      @Test
32      public void testTomcatArtifact() {
33          TomcatArtifact tomcat = new TomcatArtifact();
34          Assert.assertEquals(TomcatArtifact.TOMCAT_GROUPID, tomcat.getGroupId());
35          Assert.assertEquals(TomcatArtifact.TOMCAT_ARTIFACTID, tomcat.getArtifactId());
36          Assert.assertEquals(TomcatArtifact.DEFAULT_TOMCAT_VERSION, tomcat.getVersion());
37          Assert.assertEquals("zip", tomcat.getType());
38          Assert.assertNull(tomcat.getClassifier());
39      }
40  
41      @Test
42      public void testSpecialVersion() {
43          String version = "7.0.30";
44          TomcatArtifact tomcat = new TomcatArtifact(version);
45          Assert.assertEquals(version, tomcat.getVersion());
46      }
47  
48      @Test
49      public void testCoordinates() {
50          TomcatArtifact tomcat = new TomcatArtifact();
51          Assert.assertEquals("com.googlecode.t7mp:tomcat:zip:7.0.27", tomcat.getArtifactCoordinates());
52      }
53  
54      @Test
55      public void testCoordinatesWithSystemPath() {
56          TomcatArtifact tomcat = new TomcatArtifact();
57          tomcat.setSystemPath("/home/jbellmann/test/tomcat.zip");
58          Assert.assertEquals("/home/jbellmann/test/tomcat.zip", tomcat.getArtifactCoordinates());
59      }
